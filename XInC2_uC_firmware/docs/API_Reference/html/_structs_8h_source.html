37 /**
38  * \brief Sets the directionality of the pin encapsulated by the GlobalPin to be
39  * either output or input.
40  * \param [in] glo_pin Pointer to a GlobalPin struct
41  * \param [in] dir Desired pin directionality
42  */
43 SXC_INLINE void globalPin_set_dir(enum GlobalPin_dir dir,
44  GlobalPin const * glo_pin)
45 {
46  uint16_t curr_conf = gpio_get_config(glo_pin->port);
47  switch(dir) {
48  case PinDir_Input:
49  gpio_set_config(curr_conf & ~(glo_pin->pin << 8), glo_pin->port);
50  break;
51  case PinDir_Output:
52  gpio_set_config(curr_conf | (glo_pin->pin << 8), glo_pin->port);
53  break;
54  }
55 }

57 /**
58  * \brief Returns ON (1) if the pin encapsulated by glo_pin is logically high,
59  * else returns OFF (0). Logically high refers to electrically high if the pin
60  * is active-high, else refers to electrically low if the pin is active-low.
61  * Behaviour is UNDEFINED if the pin is configured to be an output.
62  * \param [in] glo_pin Pointer to GlobalPin struct
63  * \return ON (1) if input pin is logic-high, 0 else; garbage value if pin is an
64  * output
65  */
66 SXC_INLINE enum PinLogicState globalPin_read(GlobalPin const * glo_pin)
67 {
68  if (glo_pin->polar == Polar_ActiveHigh)
69  return (gpio_read(glo_pin->port) & glo_pin->pin) != 0;
70  else
71  return (gpio_read(glo_pin->port) & glo_pin->pin) == 0;
72 }

87 /**
88  * \brief Sets or clears the pin encapsulated by glo_pin. Pin is set logically
89  * high if input is non-zero (ON); if input is zero (OFF), pin is set logically
90  * low. Logically low/high correlate to electrically low/high only if the pin
91  * is ActiveHigh; else, they correlate to the opposite.
92  * Undefined behaviour occurs if the pin is configured to be an input.
93  * \param [in] input If non-zero, pin is set high; else, pin is set low.
94  * \param [in] glo_pin Pointer to GlobalPin struct
95  */
96 SXC_INLINE void globalPin_write(enum PinLogicState input,
97  GlobalPin const * glo_pin)
98 {
99  uint16_t or_pins = (input && glo_pin->polar == Polar_ActiveHigh) ||
100  (!input && glo_pin->polar == Polar_ActiveLow);
101  if (or_pins)
102  gpio_write(gpio_read(glo_pin->port) | glo_pin->pin, glo_pin->port);
103  else
104  gpio_write(gpio_read(glo_pin->port) & ~glo_pin->pin, glo_pin->port);
105 }

107 /**
108  * \brief Sets or clears the pin encapsulated by glo_pin. Pin is set electrically
109  * high if input is non-zero; if input is zero, pin is set electrically
110  * low.
111  * Undefined behaviour occurs if the pin is configured to be an input.
112  * \param [in] input If non-zero, pin is set high; else, pin is set low.
113  * \param [in] glo_pin Pointer to GlobalPin struct
114  */
115 SXC_INLINE void globalPin_write_raw(uint16_t input, GlobalPin const * glo_pin)
116 {
117  if (input)
118  gpio_write(gpio_read(glo_pin->port) | glo_pin->pin, glo_pin->port);
119  else
120  gpio_write(gpio_read(glo_pin->port) & ~glo_pin->pin, glo_pin->port);
121 }
