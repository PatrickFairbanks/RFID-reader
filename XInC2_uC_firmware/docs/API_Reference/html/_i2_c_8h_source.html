36 SXC_INLINE void i2c_set_mode(enum I2C_Mode mode,

44  globalPin_set_dir(PinDir_Input, &pins->scl);
45  break;
46  }
47  globalPin_set_dir(PinDir_Input, &pins->sda);
48 }

84 /**
85  * \brief Writes an 8-bit byte (the LSB of byte) onto the I2C line. Returns 1 if
86  * an ACK is read on the line after writing, returns 0 else. May only be called
87  * if the device is set as an I2C master; otherwise, undefined behaviour occurs.
88  * \param [in] byte 16-bit word whose LSB is written onto the I2C line. MSB is
89  * ignored.
90  * \param [in] pins Pointer to the I2C_Pin_Pair of interest
91  * \return 1 if an ACK is read on the line after writing, 0 else
92  */
93 SXC_INLINE uint16_t i2c_write_byte(uint16_t byte,
94  I2C_Pin_Pair const * pins)
95 {
96  for (int16_t i = 7; i >= 0; --i) {
97  if(_rol(1, i) & byte) {
98  globalPin_set_dir(PinDir_Input, &pins->sda);
99  }
100  else {
101  globalPin_set_dir(PinDir_Output, &pins->sda);
102  globalPin_write_raw(0, &pins->sda);
103  }
104  sys_clock_wait(pins->timeout);
105 
106  globalPin_set_dir(PinDir_Input, &pins->scl);
107  sys_clock_wait(pins->timeout);
108  sys_clock_wait(pins->timeout);
109  globalPin_set_dir(PinDir_Output, &pins->scl);
110  globalPin_write_raw(0, &pins->scl);
111  }
112 
113  globalPin_set_dir(PinDir_Input, &pins->sda);
114  sys_clock_wait(pins->timeout);
115  globalPin_set_dir(PinDir_Input, &pins->scl);
116  sys_clock_wait(pins->timeout);
117 
118  while(!globalPin_read_raw(&pins->scl));
119 
120  uint16_t retVal = !globalPin_read_raw(&pins->sda);
121  sys_clock_wait(pins->timeout);
122  globalPin_set_dir(PinDir_Output, &pins->scl);
123  globalPin_write_raw(0, &pins->scl);
124  sys_clock_wait(pins->timeout);
125  return retVal;
126 }

128 /**
129  * \brief Reads an 8-bit byte from the I2C line. May only be called if the device
130  * is set as an I2C master; else, undefined behaviour occurs.
131  * \param [in] nack If non-zero, respond with a "NACK" on the line after reading.
132  * \param [in] pins Pointer to the I2C_Pin_Pair of interest
133  * \return The 8-bit byte in the LSB of a 16-bit word
134  */
135 SXC_INLINE uint16_t i2c_read_byte(uint16_t nack,
136  I2C_Pin_Pair const * pins)
137 {
138  uint16_t byte = 0;
139  for (int16_t i = 7; i >= 0; i--) {
140  globalPin_set_dir(PinDir_Input, &pins->scl);
141  while(!globalPin_read_raw(&pins->scl));
142  sys_clock_wait(pins->timeout);
143  byte += byte + globalPin_read_raw(&pins->sda);
144  globalPin_set_dir(PinDir_Output, &pins->scl);
145  globalPin_write_raw(0, &pins->scl);
146  sys_clock_wait(pins->timeout);
147  }
148 
149  if (nack) {
150  globalPin_set_dir(PinDir_Input, &pins->sda);
151  }
152  else {
153  globalPin_set_dir(PinDir_Output, &pins->sda);
154  globalPin_write_raw(0, &pins->sda);
155  }
156 
157  sys_clock_wait(pins->timeout);
158  globalPin_set_dir(PinDir_Input, &pins->scl);
159  sys_clock_wait(pins->timeout);
160 
161  while(!globalPin_read_raw(&pins->scl));
162 
163  sys_clock_wait(pins->timeout);
164  globalPin_set_dir(PinDir_Output, &pins->scl);
165  globalPin_write_raw(0, &pins->scl);
166  sys_clock_wait(pins->timeout);
167  globalPin_set_dir(PinDir_Input, &pins->sda);
168  return byte;
169 }

171 /**
172  * \brief Writes an 8-bit byte (the LSB of byte) to the device with the specified
173  * address. Returns 1 if the operation succeeds and is ACKed at all stages; 0
174  * else. Function may only be called if the device is set as an I2C master;
175  * otherwise, undefined behaviour occurs.
176  * \param [in] dest_addr 16-bit word whose LSB is the I2C address to write to.
177  * \param [in] byte 16-bit word whose LSB is written. MSB is ignored.
178  * \param [in] pins Pointer to the I2C_Pin_Pair of interest
179  * \return 1 if all expected ACKs are read on the line after writing, 0 else
180  */
181 SXC_INLINE uint16_t i2c_write_to_addr(
182  uint16_t dest_addr, uint16_t byte, I2C_Pin_Pair const * pins)
183 {
184  i2c_write_start(pins);
185 
186  if (!i2c_write_byte(dest_addr, pins))
187  return 0;
188  if (!i2c_write_byte(byte, pins))
189  return 0;
190 
191  i2c_write_stop(pins);
192 
193  return 1;
194 }

238  * \param [in] pins Pointer to the I2C_Pin_Pair of interest
239  * \return 1 if all expected ACKs are read on the line after writing, 0 else
240  */
241 SXC_INLINE uint16_t i2c_write_buf_to_register(
242  uint16_t dest_addr, uint16_t dest_reg, uint16_t* buf, size_t len,
243  I2C_Pin_Pair const * pins)
244 {
245  i2c_write_start(pins);
246 
247  if (!i2c_write_byte(dest_addr, pins))
248  return 0;

250  return 0;

253  return 0;

257 
258  return 1;
