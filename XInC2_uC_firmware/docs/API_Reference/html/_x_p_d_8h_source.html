58 /**
59  * \brief Checks the status of the XPD's connectivity.
60  * \return XPD connection status. 1 implies connected, 0 implies not.
61  */
62 SXC_INLINE uint16_t xpd_conn_status(void)
63 {
64 #ifndef __XINC2__
65 #else
66  register int reg1Out __asm__("R1");
67  register int reg1In __asm__("R1") = 0x4000;
68  register int reg2In __asm__("R2") = -1;
69  ASM_VOLATILE("jsr\tr6, XPD_ShiftInOut"
70  : "=r"(reg1Out)
71  : "r"(reg1In), "r"(reg2In)
72  : "%R6");
73  if ((reg1Out & 0x3FFF) != 1) return 0;
74 #endif
75  return 1;
76 }

105 /**
106  * \brief Prints an integer out on the XPD as a formatted string. Formatting
107  * is dependent on the echo_flag passed in.
108  * \param [in] number integer to print
109  * \param [in] type echo_flag describing the formatting to use
110  */
111 SXC_INLINE void xpd_echo_int(int number, enum echo_flag type)
112 {
113  switch (type) {
114  case XPD_Flag_SignedDecimal: {
115  _MaskCallIn(unsigned int, R1, number, XPD_EchoSignedDec);
116  break;
117  }
118  case XPD_Flag_UnsignedDecimal: {
119  _MaskCallIn(unsigned int, R1, number, XPD_EchoUnsignedDec);
120  break;
121  }
122  case XPD_Flag_Hex: {
123  _MaskCallIn(unsigned int, R1, number, XPD_EchoHex);
124  break;
125  }
126  case XPD_Flag_HexNoPrefix: {
127  _MaskCallIn(unsigned int, R1, number, XPD_EchoHexNP);
128  break;
129  }
130  default:
131  break;
132  }

173 /**
174  * \brief Reads a single 8-bit byte from the input line of the XPD. Returns when
175  * a byte is received or when "timeout" attempts
176  * \return The raw 8-bit byte in the LSB of a signed integer. If the read timed
177  * out, then 0xFFFF is returned instead.
178  */
179 SXC_INLINE int xpd_getchar_timeout(int timeout)
180 {
181 #ifndef __XINC2__
182  int buf;
183  int xin_dupl = dup(fileno(xin));
184  fcntl(xin_dupl, F_SETFL, O_NONBLOCK);
185  int err = read(xin_dupl, &buf, 1);
186  if (err == EAGAIN) {
187  sleep(timeout);
188  read(xin_dupl, &buf, 1);
189  }
190  close(xin_dupl);
191  if (err <= 0)
192  return -1;
193  return buf;
194 #else
195  _MaskCallInOut(int, R1, timeout, out, XPD_ReadByteWithTimeout);
196  if (out < 0)
197  return out; // This has been replaced with the return value
198  else
199  return out & 0xFF;
200 #endif
201 }
