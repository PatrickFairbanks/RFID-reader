39 /**
40  * \brief Packs the bits of the lower byte of the input as described by the mask
41  * in the upper byte of the input. See details, or XInC2 manual for more.
42  * \param[in] input an arbitrary 16-bit number
43  * \details
44  * The 8-bit mask in the upper byte determines which bits to select from the
45  * 8-bit value in the lower byte for the output. Only value bits whose positions
46  * align with bits set to 1 in the mask are selected. These bits are then
47  * packed, right-aligned, into as many bits as there were '1's in the mask.
48  * \return The mask-selected bits of the lower byte, packed right-aligned. All
49  * other bits are set to 0.
50  *
51  */
52 SXC_INLINE uint16_t sfu_pack_bits(uint16_t input)
53 {
54 #ifndef __XINC2__
55  pthread_mutex_lock(&SFUpack_lock);
56  mock_mem_map[SFUpack] = input;
57  uint16_t mask = (input & 0xFF00) >> 8;
58  uint16_t val = input & 0x00FF;
59  uint16_t result = 0;
60  uint16_t i;
61  for (i=0; mask; mask >>= 1) {
62  if (mask & 0x1) {
63  result |= ((val & 0x1) << i);
64  ++i;
65  }
66  val >>= 1;
67  }
68  mock_mem_map[SFUpack] = result;
69  pthread_mutex_unlock(&SFUpack_lock);
70  return result;
71 #else
72  _outp(input, SFUpack);
73  return _inp(SFUpack);
74 #endif
75 }

77 /**
78  * \brief The bit position of the least significant bit set to 1 is returned as
79  * a 4-bit integer in the rightmost bits of the return value. If no bit of
80  * input is set to 1, the leftmost bit (15) of the return value is set to 1.
81  * (zero flag).
82  * \param[in] input an arbitrary 16-bit number
83  * \return The 4-bit integer representing the first bit set to 1 in input.
84  * If input is 0, returns 0x8000.
85  */
86 SXC_INLINE uint16_t sfu_least_significant_one(uint16_t input)
87 {
88 #ifndef __XINC2__
89  pthread_mutex_lock(&SFUls1_lock);
90  mock_mem_map[SFUls1] = input;
91  uint16_t result;
92  if (!input)
93  result = (1 << (16-1));
94  else {
95  uint16_t i;
96  for (i=0; !(input & 0x1); input >>= 1) //while bit[i] == 0
97  ++i;
98  result = i;
99  }
100  mock_mem_map[SFUls1] = result;
101  pthread_mutex_unlock(&SFUls1_lock);
102  return result;
103 #else
104  _outp(input, SFUls1);
105  return _inp(SFUls1);
106 #endif
107 }

109 /**
110  * \brief Returns the number of bits set to 1 in input. Value is returned as a
111  * 5-bit integer in the rightmost bits of the SFUpop register.
112  * \param[in] input an arbitrary 16-bit number
113  * \return The 5-bit integer representing the number of bits set to 1 in input.
114  */
115 SXC_INLINE uint16_t sfu_pop_count(uint16_t input)
116 {
117 #ifndef __XINC2__
118  pthread_mutex_lock(&SFUpop_lock);
119  mock_mem_map[SFUpop] = input;
120  uint16_t result;
121  for (result = 0; input; input >>= 1)
122  result += (input & 0x1);
123  mock_mem_map[SFUpop] = result;
124  pthread_mutex_unlock(&SFUpop_lock);
125  return result;
126 #else
127  _outp(input, SFUpop);
128  return _inp(SFUpop);

132 /**
133  * \brief Returns the bitwise-reversed value of input, and writes it to the
134  * SFUrev register.
135  * \param[in] input an arbitrary 16-bit number
136  * \return The input number, bitwise-reversed.
137  */
138 SXC_INLINE uint16_t sfu_reverse_bits(uint16_t input)
139 {
140 #ifndef __XINC2__
141  pthread_mutex_lock(&SFUrev_lock);
142  mock_mem_map[SFUrev] = input;
143  uint16_t result = input; //Start with output MSB(after shifts)=input LSB
144  uint16_t post_shifts = 16-1;
145  for (input >>= 1; input; input >>=1) {
146  result <<= 1;
147  result |= (input & 0x1);
148  post_shifts--;
149  }
150  result <<= post_shifts;
151  mock_mem_map[SFUrev] = result;
152  pthread_mutex_unlock(&SFUrev_lock);
153  return result;
154 #else
155  _outp(input, SFUrev);
156  return _inp(SFUrev);
157 #endif
158 }
