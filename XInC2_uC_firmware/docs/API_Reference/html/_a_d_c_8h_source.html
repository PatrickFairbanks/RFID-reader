38 /**
39  * \brief Powers the ADC on. Does nothing if the ADC is already powered on. It
40  * is recommended to leave the ADC on only when necessary due to high power
41  * consumption.
42  */
43 SXC_INLINE void ADC_power_on(void)
44 {
45 #ifndef __XINC2__
46  mock_mem_map[ADCcfg0] |= ADC_ON_BIT;
47 #else
48  uint16_t cfg0 = _inp(ADCcfg0)|ADC_ON_BIT;
49  uint16_t cfg1 = _inp(ADCcfg1);
50  _outp(cfg0, ADCcfg0);
51  _outp(cfg0|ADC_START_CONV_BIT, ADCcfg0);
52  _outp(cfg1|ADC_CFG1_RESET_BIT, ADCcfg1);
53  //NOPs because ADC reset bit must be on for at least 3 clock cycles
54  _nop();
55  _nop();
56  _nop();
57  _outp(cfg1&~ADC_CFG1_RESET_BIT, ADCcfg1);
58 #endif
59 }

100 /**
101  * \brief Sets the 2-bit channel number(s) for the ADC to read from. Reads in 1
102  * channel number if the ADC is in single conversion mode. If it is in
103  * continuous sampling mode, then as many channel numbers are read in as the ADC
104  * is configured for (3 or 4). Only the least significant 2 bits from each number
105  * are read; all others are ignored. No array boundary checking is done.
106  * Behaviour is unpredictable if the ADC is not configured before calling this
107  * function.
108  * \param [in] nums Pointer to desired sampling channel numbers
109  */
110 SXC_INLINE inline void ADC_set_channel_nums(uint16_t const * nums)
111 {
112  uint16_t cfg = ADC_get_config() & 0xFF00; //clear the channel nums
113  size_t bound = (!(cfg & ADC_CONT_SAMPLING)) ? 1 :
114  (cfg & ADC_THREE_CHANNEL_MODE) ? 3 : 4;
115  for (size_t i = 0; i < bound; ++i)
116  cfg |= _rol( nums[i]&0x3, _rol(i,1) );
117  ADC_set_config(cfg);
118 }

120 /**
121  * \brief Sets the 6-bit number of ADC clocks to wait for after a sample has been
122  * converted. This is only meaningful if the ADC is set for continuous sampling
123  * mode; this has no effect in single sampling mode.
124  * \param[in] num_clocks Number of ADC clocks to wait before re-sampling. Only
125  * the least significant 6 bits are considered; all others are ignored.
126  */
127 SXC_INLINE void ADC_set_clock_wait(uint16_t num_clocks)
128 {
129 #ifndef __XINC2__
130  mock_mem_map[ADCcfg1] &= 0xFFC0;
131  mock_mem_map[ADCcfg1] |= (num_clocks&0x003F);
132 #else
133  uint16_t cfg = _inp(ADCcfg1);
134  cfg &= 0xFFC0;
135  cfg |= (num_clocks&0x003F);
136  _outp(cfg, ADCcfg1);
137 #endif
138 }
