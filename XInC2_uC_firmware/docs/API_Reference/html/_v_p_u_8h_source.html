102 /**
103  * \brief Sets up the operation to be carried out by the VPU in further function
104  * calls. Operation is selected by bitwise-OR of configuration flags and the
105  * operand address mask. Further details in XInC2 manual.
106  * \param[in] cfg0 CFG0 register flags to set up the VPU operation.
107  * \param[in] cfg1 CFG1 regiser flags to further configure the VPU operation.
108  * \param[in] opadrmask Operand address mask. The lower byte refers to operand
109  * A's mask, and the upper one to operand B's.
110  */
111 SXC_INLINE void vpu_set_config(uint16_t cfg0, uint16_t cfg1, uint16_t opadrmask)
112 {
113 #ifndef __XINC2__
114 #else
115  _outp(cfg0, VPUcfg0);
116  _outp(cfg1, VPUcfg1);
117  _outp(opadrmask, VPUopadrmsk);
118 #endif
119 }

121 /**
122  * \brief Returns the value (configuration) in the chosen VPU register.
123  * \param[in] reg Desired VPU register.
124  * \return The desired VPU register's configuration. 0 if invalid input argument.
125  */
126 SXC_INLINE uint16_t vpu_get_config(enum VPU_config_register reg)
127 {
128  safe_mode_assert(reg < kNumVPUCfgReg);
129  uint16_t addr = 0;
130  switch(reg) {
131  case VPU_CFG0:
132  addr = VPUcfg0;
133  break;
134  case VPU_CFG1:
135  addr = VPUcfg1;
136  break;
137  case VPU_ADDRMASK:
138  addr = VPUopadrmsk;
139  break;
140  default:
141  break;
142  }
143 #ifndef __XINC2__
144  return mock_mem_map[addr];
145 #else
146  return _inp(addr);
147 #endif
148  return 0;
149 }

151 /**
152  * \brief Executes the operation configured in the VPU after vpu_setup is
153  * called, with the specified operands. Further details in XInC2 manual.
154  * \detailed All arguments must be pre-allocated. No bounds checking or
155  * dynamic memory allocation is performed: BE VERY MEMORY-WARY.
156  * \param[in] opA Array containing first operand values.
157  * \param[in] opA_len Length of the aforementioned array.
158  * \param[in] opB Array containing second operand values.
159  * \param[in] opB_len Length of the aforementioned array.
160  * \param[in] res Array to store the results of the vector operation.
161  * \param[in] res_len Length of the aforementioned array.
162  * \return Returns res (pointer to first element of result array).
163  */
164 SXC_INLINE uint16_t* vpu_start(uint16_t *opA, uint16_t opA_len,
165  uint16_t *opB, uint16_t opB_len,
166  uint16_t *res, uint16_t res_len)
167 {
168 #ifndef __XINC2__
169 #else
170  _outp( ((opA_len - 1) << 8)|(opB_len - 1), VPUopadrmsk); //pure magic, no docs
171  _outp( res_len - 1, VPUrsadrsmk);
172  _outp((uint16_t)opA, VPUopAba);
173  _outp((uint16_t)opB, VPUopBba);
174  _outp((res_len << 8)|1, VPUfilterLength); //not sure why the outer loop len is this
175  uint16_t result_addr = (uint16_t)res;
176  _outp(result_addr, VPUrsba);
177  _outp(result_addr, VPUopA); //no idea what is with these last 3 lines:
178  //some sort of micro-optimization
179 #endif
180 }
