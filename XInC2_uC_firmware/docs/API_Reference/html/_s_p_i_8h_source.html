50 /**
51  * \brief Returns the standard SPI configuration value for the given system
52  * frequency and SPI.
53  * \param[in] sf input system frequency
54  * \param[in] name SPI to target
55  * \return The SPI config for valid frequencies. 0 else.
56  */
57 SXC_INLINE uint16_t get_SPI_std_cfg(enum sys_freq sf, enum SPI_name name)
58 {
59  safe_mode_assert(name < kNumSPIs);
60  switch(name) {
61  case SPI0:
62  switch(sf) {
63  case _12_288_MHz:
64  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_16;
65  case _24_576_MHz:
66  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_32;
67  case _36_864_MHz:
68  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_64;
69  case _49_152_MHz:
70  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_64;
71  case _73_728_MHz:
72  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_128;
73  case _98_304_MHz:
74  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_128;
75  case _110_592_MHz:
76  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_256;
77  }
78  case SPI1:
79  switch(sf) {
80  case _12_288_MHz:
81  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_2;
82  case _24_576_MHz:
83  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_4;
84  case _36_864_MHz:
85  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_4;
86  case _49_152_MHz:
87  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_8;
88  case _73_728_MHz:
89  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_8;
90  case _98_304_MHz:
91  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_16;
92  case _110_592_MHz:
93  return SPI_ENABLE|SPI_MASTER|SPI_CLK_RATE_DIV_16;
94  }
95  }
96  return 0;

169 /**
170  * \brief Writes an 8-bit byte onto the MOSI or MISO line of the selected SPI,
171  * depending on whether the SPI is configured as a master or slave, respectively.
172  * \param[in] val LSB: The 8-bit value to write onto the line. MSB: ignored
173  * \param[in] name The selected SPI.

175 SXC_INLINE void SPI_write(uint16_t val, enum SPI_name name)

177 #ifndef __XINC2__
178 #else
179  uint16_t out_reg = get_spi_addr(SPI_TX, name);
180  uint16_t in_reg = get_spi_addr(SPI_RX, name);
181  _outp(val, out_reg);
182  _inp(in_reg);
183 #endif
184 }

186 /**
187  * \brief Reads an 8-bit byte from the MOSI or MISO line of the selected SPI,
188  * depending on whether the SPI is configured as a slave or master, respectively.
189  * \param[in] name The selected SPI.
190  * \return LSB: The 8-bit value on the line at the time of function call. MSB:
191  * undefined
192  */
193 SXC_INLINE uint16_t SPI_read(enum SPI_name name)
194 {
195 #ifndef __XINC2__
196  return 0; //dummy
197 #else
198  uint16_t in_reg = get_spi_addr(SPI_RX, name);
199  uint16_t out_reg = get_spi_addr(SPI_TX, name);
200  _outp(0, out_reg); //dummy write with 0 so an accidental command isn't sent
201  return _inp(in_reg);
202 #endif
203 }
