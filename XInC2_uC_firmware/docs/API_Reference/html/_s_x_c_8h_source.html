63 SXC_INLINE uint16_t _rol(uint16_t a, uint16_t b)
64 {
65 #ifdef __USE_ROL
66  int retval;
67  __asm__("rol\t%0,%1,%2" : "=r"(retval) : "r"(a), "rn"(b) :);
68  return retval;
69 #else
70  return a << b;
71 #endif
72 }
