39 SXC_INLINE void varactor_setup(enum sys_freq sf)
40 {
41  //The 0x1 should do nothing (read-only bit), but it was there in the original
42  timer_set_config_ccm(
43  0x1 | TIMER_CCM_COMP_NOT_CAP | TIMER_CCM_CMP_OUT_EQ_1 | TIMER_CCM_OUT_ENABLE
44  | TIMER_CCM_OUTPUT_RESET_SET, 1, TIMER_B
45  );
46  timer_write_CCR(500, 0, TIMER_B); //why 500? nfi
47  timer_write_CCR(190, 1, TIMER_B); //why 190? nfi
48  timer_set_config(get_tmrbcfg(sf), TIMER_B);
49 }
