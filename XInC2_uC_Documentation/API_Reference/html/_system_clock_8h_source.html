159 SXC_INLINE uint16_t get_pllcfg0(enum crystal_freq cf, enum sys_freq sf)
160 {
161  switch(cf) {
162  case crys_12_288_MHz:
163  switch(sf) {
164  case _24_576_MHz:
165  return PLL_ENABLE|(0x2 << PLL_DIVIDER_BITSHIFT)|
166  (0x8 << PLL_FEEDBACK_DIVIDER_BITSHIFT);
167  case _49_152_MHz:
168  return PLL_ENABLE|(0x2 << PLL_DIVIDER_BITSHIFT)|
169  (0x10 << PLL_FEEDBACK_DIVIDER_BITSHIFT);
170  case _73_728_MHz:
171  return PLL_ENABLE|(0x2 << PLL_DIVIDER_BITSHIFT)|
172  (0x18 << PLL_FEEDBACK_DIVIDER_BITSHIFT);
173  default:
174  return 0;
175  }
176 
177  case crys_24_576_MHz:
178  switch(sf) {
179  case _36_864_MHz:
180  return PLL_ENABLE|(0x1 << PLL_DIVIDER_BITSHIFT)|
181  (0x3 << PLL_FEEDBACK_DIVIDER_BITSHIFT);
182  case _49_152_MHz:
183  return PLL_ENABLE|(0x1 << PLL_DIVIDER_BITSHIFT)|
184  (0x2 << PLL_FEEDBACK_DIVIDER_BITSHIFT);
185  case _73_728_MHz:
186  return PLL_ENABLE|(0x2 << PLL_DIVIDER_BITSHIFT)|
