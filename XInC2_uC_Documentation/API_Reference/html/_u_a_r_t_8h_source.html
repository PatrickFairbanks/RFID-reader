25 /**
26  * \brief Transmits the LSB of the input word as a byte over a UART tx pin.
27  * \param [in] byte 16-bit word whose LSB is transmitted. MSB is ignored.
28  * \param [in] pins Pointer to the UART_Pin_Pair of interest
29  */
30 SXC_INLINE void uart_write_byte(uint16_t byte, UART_Pin_Pair const * pins)
31 {
32  globalPin_write_raw(0, &pins->tx); //write the start bit
33  sys_clock_wait(pins->timeout);
34  //Write bits, starting with lsb
35  for (int16_t i = 0; i <= 7; ++i) {
36  if (_rol(1, i) & byte)
37  globalPin_write_raw(1, &pins->tx);
38  else
39  globalPin_write_raw(0, &pins->tx);
40  sys_clock_wait(pins->timeout);
41  }
42  globalPin_write_raw(1, &pins->tx); //write the stop bit
43  sys_clock_wait(pins->timeout);
44 }

46 /**
47  * \brief Reads an 8-bit byte from the rx pin of a UART pair.
48  * \param [in] pins Pointer to the UART_Pin_Pair of interest
49  * \return 8-bit byte read from the UART rx, in the LSB of a 16-bit word
50  */
51 SXC_INLINE uint16_t uart_read_byte(UART_Pin_Pair const * pins)
52 {
53  uint16_t byte = 0;
54  while(globalPin_read_raw(&pins->rx)); //wait until the start bit is written (0)
55  sys_clock_wait(pins->timeout);
56  //Read bits, starting with lsb
57  for (int16_t i = 0; i <= 7; ++i) {
58  byte += _rol(globalPin_read_raw(&pins->rx), i);
59  sys_clock_wait(pins->timeout);
60  }
61  while(!globalPin_read_raw(&pins->rx)); //wait until stop bit is written (1)
62  return byte;
63 }
