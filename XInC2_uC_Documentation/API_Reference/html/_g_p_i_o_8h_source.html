35 /**
36  * \brief Configures the selected GPIO port pins' directionality based on the
37  * provided configuration. CLOBBERS THE OUTPUT BUFFER REGISTER (LSB). Call
38  * gpio_set_config((gpio_get_config() & 0xFF) | cfg) to avoid this, if you care.
39  * \param[in] cfg Desired pin configuration. Lower byte clobbers output buffer
40  * register. Set bits in the upper byte cause that pin to be an output;
41  * cleared bits correlate to input pins. Because this is the MSB,
42  * bit8 of the entire word corresponds to pin0, bit9 to pin1, and so on.
43  * \param[in] port The selected GPIO port.
44  */
45 SXC_INLINE void gpio_set_config(uint16_t cfg, enum GPIO_port_name port)
46 {
47  safe_mode_assert(port < kNumGPIOPorts);
48 #ifndef __XINC2__
49 #else
50  switch(port){
51  case GPIO_A:
52  _outp(cfg, GPAcfg);
53  break;
54  case GPIO_B:
55  _outp(cfg, GPBcfg);
56  break;
57  case GPIO_C:
58  _outp(cfg, GPCcfg);
59  break;
60  case GPIO_D:

62  break;
63  case GPIO_E:

214 /**
215  * \brief Atomically sets the desired pin in the desired port to 1.
216  * \param[in] pin_num The pin to set
217  * \param[in] port The GPIO port to target.
218  * \param[in] sem_num Semaphore to acquire for atomicity.
219  */
220 SXC_INLINE void gpio_atomic_pin_set(uint16_t pin_num, enum GPIO_port_name port,
221  uint16_t sem_num)
222 {
223  sem_lock(sem_num);
224  uint16_t value = gpio_read(port);
225  gpio_write(value|(1 << pin_num), port);
226  sem_unlock(sem_num);
227 }
