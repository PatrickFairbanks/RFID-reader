40 /**
41  * \brief Starts running the selected thread.
42  * No effect if the thread is already running.
43  * \param[in] thrd_num 3-bit integer representing which thread to run.
44  */
45 SXC_INLINE void thread_run(uint16_t thrd_num)
46 {
47  safe_mode_assert(thrd_num < kNumThrds);
48  sem_lock(SEM_THRD);
49  thrds_vector &= ~(1 << thrd_num);
50 #ifndef __XINC2__
51 #else
52  _outp(thrds_vector, SCUstop);
53 #endif
54  sem_unlock(SEM_THRD);
55 }

74 /**
75  * \brief Stops/starts the execution of the selected threads. Multiple threads
76  * may be affected simultaneously. Setting bitN stops threadN, while clearing it
77  * starts/continues running the thread. It is not recommended to call this
78  * function, unless one knows the run status of all threads and preserves the
79  * status of threads which are not intended to be run or stopped by a call to
80  * this function.
81  * \param[in] thrds Lower byte: set bits correspond to threads to stop, cleared
82  * bits correspond to threads to run. Upper byte is irrelevant.
83  */
84 SXC_INLINE void thread_stoprun_multiple(uint16_t thrds)
85 {
86  sem_lock(SEM_THRD);
87 #ifndef __XINC2__
88 #else
89  thrds_vector = thrds;
90  _outp(thrds_vector, SCUstop);
91 #endif
92  sem_unlock(SEM_THRD);
93 }

110 /**
111  * \brief Sets up initial arguments and initial function to execute in the
112  * selected thread. Does NOT actually start the thread. Behaviour is undefined
113  * if the thread is already executing.
114  * \param[in] func Function pointer to the input function. Function may have no
115  * return value and no inputs, or pointers for either or both.
116  * \param[in] ptr Pointer to input arguments to provide to thread function.
117  * \param[in] thrd_num The thread to set up.
118  */
119 SXC_INLINE void thread_setup(void* (*func)(void*), void* ptr, uint16_t thrd_num)
120 {
121  safe_mode_assert(thrd_num < kNumThrds);
122  sem_lock(SEM_THRD);
123 #ifndef __XINC2__
124 #else
125  _outp((thrd_num << THREAD_PC_SHIFT)|THREAD_R2, SCUpntr);
126  _outp((uint16_t)ptr, SCUreg);
127  _outp((uint16_t)func, SCUpc);
128 #endif
129  sem_unlock(SEM_THRD);
130 }

132 /**
133  * \brief Writes the 16-bit return value pointed to by value_p to the top of the
134  * calling thread's stack, then stops execution of the thread.
135  * \param [in] value_p pointer to desired return value
136  */
137 SXC_INLINE void thread_return(void *value_p)
138 {
139 #ifndef __XINC2__
140  pthread_t thrd_num = pthread_self();
141 #else
142  uint16_t thrd_num = _thrd();
143  // This will fall apart if the stacks are not all the same size
144  uint16_t* stack_top = THREAD0_STACK + thrd_num*_DEFAULT_STACK_SIZE;
145  *stack_top = *(uint16_t*)value_p;
146 #endif
147  thread_stop(thrd_num);
148  return;
149 }

151 /**
152  * \brief Returns the 16-bit value at the top of the stack of the target thread.
153  * Assuming that the thread called thread_return and was not restarted before
154  * calling this function, this will return the value that thread_return was
155  * called with; otherwise, the value is garbage.
156  * \param [in] thrd_num Thread whose return value to read
157  * \return The 16-bit value at the top of the stack of the target thread
158  */
159 SXC_INLINE uint16_t thread_read_retval(uint16_t thrd_num)
160 {
161  safe_mode_assert(thrd_num < kNumThrds);
162  sem_lock(SEM_THRD);
163  // This will fall apart if the stacks are not all the same size
164  uint16_t retval = THREAD0_STACK[thrd_num*_DEFAULT_STACK_SIZE];
165  sem_unlock(SEM_THRD);
166  return retval;
167 }
