51 /**
52  * \brief Configures the selected LFSR to behave according to the provided
53  * configuration. See XInC2 manual for details.
54  * \param[in] lfsr_msb 4-bit integer representing the MSB of the LFSR. Also
55  * equal to desired LFSR length-1. Bits beyond bit3 are ignored.
56  * \param[in] data_msb 4-bit integer representing the MSB of the data to shift
57  * in. Also equal to desired data length-1. Bits beyond bit3 are ignored.
58  * \param[in] fb_mode The LFSR feedback mode to use.
59  * \param[in] tap The LFSR tap configuration to use (see hardware config)
60  * \param[in] lfsr The LFSR to configure.
61  */
62 SXC_INLINE void lfsr_set_config(uint16_t lfsr_msb, uint16_t data_msb,
63  enum LFSR_fb_mode fb, uint16_t tap, enum LFSR_name lfsr)
64 {
65  safe_mode_assert(lfsr < kNumLFSRs);
66 #ifndef __XINC2__
67 #else
68  uint16_t cfg = (lfsr_msb & 0xF) | ((data_msb & 0xF) << 4) | (fb << 8);
69  if (lfsr == LFSR_NAME_0) {
70  _outp(cfg, LFSR0cfg);
71  _outp(tap, LFSR0tapcfg);
72  }
73  else if (lfsr == LFSR_NAME_1) {
74  _outp(cfg, LFSR1cfg);
75  _outp(tap, LFSR1tapcfg);
76  }
77 #endif
78 }

80 /**
81  * \brief Returns either the selected LFSR's configuration, or its tap config.
82  * User selects which to return.
83  * \param[in] tap_flag if NOT 0, function returns the LFSR's tap config, rather
84  * than the LFSR's config.
85  * \param[in] lfsr The LFSR to target.
86  * \return The selected LFSR's configuration if tap_flag is 0, its tap config
87  * otherwise. Returns 0 if an invalid LFSR is selected.
88  */
89 SXC_INLINE uint16_t lfsr_get_config(uint16_t tap_flag, enum LFSR_name lfsr)
90 {
91  safe_mode_assert(lfsr < kNumLFSRs);
92 #ifndef __XINC2__
93 #else
94  if (lfsr == LFSR_NAME_0) {
95  if (tap_flag)
96  return _inp(LFSR0tapcfg);
97  return _inp(LFSR0cfg);
98  }
99  else if (lfsr == LFSR_NAME_1) {
100  if (tap_flag)
101  return _inp(LFSR1tapcfg);
102  return _inp(LFSR1cfg);
103  }
104 #endif
105  return 0;
106 }

108 /**
109  * \brief Reads the value stored/being shifted out in the selected LFSR,
110  * depending on the desired access mode.
111  * \param[in] mode Flag specifying whether to read all of the bits in the
112  * chosen LFSR, or just the ones being sequentially shifted out.
113  * \param[in] lfsr The LFSR to target.
114  * \return The selected LFSR stored value in the selected access mode. Returns
115  * 0 for an invalid LFSR or an invalid access mode.
116  */
117 SXC_INLINE uint16_t lfsr_read(enum LFSR_access_mode mode, enum LFSR_name lfsr)
118 {
119  safe_mode_assert(lfsr < kNumLFSRs);
120  safe_mode_assert(mode < kNumLFSRAccessModes);
121 #ifndef __XINC2__
122 #else
123  if (lfsr == LFSR_NAME_0) {
124  if (mode == LFSR_ACCESS_ALL)
125  return _inp(LFSR0reg);
126  else if (mode == LFSR_ACCESS_SEQUENTIAL)
127  return _inp(LFSR0data);
128  }
129  else if (lfsr == LFSR_NAME_1) {
130  if (mode == LFSR_ACCESS_ALL)
131  return _inp(LFSR1reg);
132  else if (mode == LFSR_ACCESS_SEQUENTIAL)
133  return _inp(LFSR1data);
134  }
135 #endif
136  return 0;
137 }
