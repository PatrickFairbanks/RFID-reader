158 //allows the user to setup a BBU of their choosing without worrying about registers
159 SXC_INLINE void bbu_setup(enum BaudRate br, enum BBUMode mode, enum BBU_name name,
160  uint16_t start_word)
161 {
162  safe_mode_assert(name < kNumBBUs);
163 //#ifndef __XINC2__
164 //#else
165  switch(mode) {
166  case BitStream:
167  _outp(BBU_BIDIRECTION_ENABLE|BBU_ENABLE|BBU_ROM_ENABLE|BBU_TX_ENABLE,
168  get_bbu_address(BBU_CFG0, name));
169  _outp(start_word, get_bbu_address(BBU_start, name));
170  _outp(br, get_bbu_address(BBU_brg, name));
171  break;
172  case SlowClock:
173  _outp(0, get_bbu_address(BBU_CFG0, name));
174  _outp(br, get_bbu_address(BBU_brg, name));
175  break;
176  default:
177  break;
178  }
179 //#endif
180 }
